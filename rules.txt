rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 1. KULLANICI ÖZEL VERİLERİ (GÜVENLİ BÖLGE)
    // Sadece kullanıcı kendisi veya Admin kullanicilar koleksiyonunu okuyup yazabilir.
    match /kullanicilar/{userId} {
      allow read, write: if request.auth != null && (request.auth.uid == userId || request.auth.token.email == 'mremzi@gmail.com');
      
      match /kovanlar/{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      match /sonenKovanlar/{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      match /bosKovanlar/{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      // Yeni eklenen koleksiyonlar: envanter ve gorevler
      match /envanter/{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
      match /gorevler/{document=**} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }

    // 2. DUYURU PANOSU (GENEL BÖLGE)
    // Kural: Herkes okuyabilir, Sadece Admin (mremzi@gmail.com) yazabilir.
    match /pano/{docId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.email == 'mremzi@gmail.com';
    }
    
  }
}
