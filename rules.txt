rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {
    
    // 1. KULLANICI ÖZEL VERİLERİ (GÜVENLİ BÖLGE)
    // Kural: "kullanicilar" klasörüne girildiğinde, kişi sadece kendi ID'sini taşıyan klasörü açabilir.
    // Başkasının verisine erişemez.
    match /kullanicilar/{userId}/kovanlar/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    match /kullanicilar/{userId}/sonenKovanlar/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }
    match /kullanicilar/{userId}/bosKovanlar/{document=**} {
      allow read, write: if request.auth != null && request.auth.uid == userId;
    }

    // 2. DUYURU PANOSU (GENEL BÖLGE)
    // Kural: Herkes okuyabilir, Sadece Admin (mremzi@gmail.com) yazabilir.
    match /pano/{docId} {
      allow read: if true;
      allow write: if request.auth != null && request.auth.token.email == 'mremzi@gmail.com';
    }
    
  }
}
